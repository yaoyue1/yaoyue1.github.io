canvas学习一点点（四）--仿知乎登录页

发表于 2017-04-05 | 分类于 canvas

canvas学习一点点的第四篇啦~，这次写的是一个仿知乎的登录页面:传送门,感觉这效果已经被n多人写(wan)过(huai)了，但是感觉这个效果还是很适合做背景的，于是我还是跟风了一把。
看完你会发现其实这个效果的写法和下雨那个很像，只不过下雪那个运动会稍微简单点，但是本质是不变的，都是控制粒子的运动。上代码



    $(function(){
      var cvs = document.getElementById(“mycanvas”);
      var ctx = cvs.getContext(‘2d’);
      var docWidth = document.documentElement.clientWidth;
      var docHeight = document.documentElement.clientHeight;
      cvs.width = docWidth;
      cvs.height = docHeight;//初始化准备，让canvas的宽高等于可视区的宽高
      var dotNum = 100;//整个区域粒子的数量
      var dots = [];//用来存放粒子对象的数组
      var mouse = new Object();//声明的对象用来存放鼠标的初始X、Y值
      mouse.x = 800;
      mouse.y = 350;
      for(var i=0;i= docHeight-obj.r){//下边界小球反弹处理
          obj.y = docHeight-obj.r;
          obj.speedY=-1;
        }
        if(obj.x<=obj.r){//左边界的小球反弹处理
          obj.x = obj.x;
          obj.speedX=-1;
        }
        if(obj.x >= docWidth-obj.r){//右边界的小球反弹处理
          obj.x = docWidth-obj.r;
          obj.speedX=-1;
        }
      }
      function ballAndball(obj1,obj2){//封装函数用来求两个球或者是球和鼠标直接的直线距离
        var disX = Math.abs(obj1.x-obj2.x);//相距的X绝对值
        var disY = Math.abs(obj1.y-obj2.y);//相距的Y绝对值
        return Math.sqrt(disXdisX+disY*disY)//平方根求直线距离
      }
      function mouseball(mouse,obj){//鼠标移动，把鼠标和距离其130内的小球连成线
        if(ballAndball(mouse,obj)<130){
          ctx.lineWidth = 0.2;
          ctx.beginPath();
          ctx.moveTo(obj.x,obj.y);
          ctx.lineTo(mouse.x,mouse.y);
          ctx.strokeStyle = obj.color;
          ctx.stroke();

    }
  }
  render();
  function render(){
    ctx.clearRect(0, 0, docWidth, docHeight);//清除画布
    for(var j= 0;j < dots.length;j++){//两层for循环，第一层是先画小球然后让每个小球移动，鼠标移动连线
      little = dots[j]
      move(little)
      drawball(little.x,little.y,little.r,little.color)
      mouseball(mouse,little)
      for(var k = 0;k < dots.length;k++){//第二层循环是用来找出两个球如果相距100就把他们用线连接起来
        if(ballAndball(dots[j],dots[k])<100){
          ctx.lineWidth = 0.2;
          ctx.globalAlpha = (130 - ballAndball(dots[j], dots[k])) * 1 / 100;//这句代码是用来控制线的消失和出现的效果
          ctx.beginPath();
          ctx.moveTo(dots[j].x, dots[j].y);
          ctx.lineTo(dots[k].x, dots[k].y);
          ctx.strokeStyle = dots[j].color;
          ctx.stroke();
        }
      }
      ctx.globalAlpha = 1.0
    }
    requestAnimationFrame(render);//浏览器自动刷新时间来循环调用render
  }


  cvs.onmousemove = function(event){//鼠标移动是的X、Y坐标
    e = event || window.event;
    mouse = new Object();
    mouse.x = e.offsetX;
    mouse.y = e.offsetY;
  }
})
